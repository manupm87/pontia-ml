# Dependencias EXTRA para entrenar, explorar y registrar experimentos.
# Instalación: pip install -r requirements.txt -r requirements-train.txt
#
# Estas librerías NO se necesitan para servir el modelo en producción
# (Render / Streamlit Cloud); por eso viven en un fichero aparte: así el
# entorno público se mantiene ligero (~350 MB en lugar de ~1 GB) y los
# tiers gratuitos no se quedan sin RAM.
#
# Cuándo necesitas instalar este fichero:
#   - Reentrenar los modelos con `python -m src.train`
#     (la red neuronal usa TensorFlow/Keras).
#   - Reejecutar la búsqueda de hiperparámetros (`python -m src.tuning`).
#   - Lanzar el módulo de balanceo de clases (`python -m src.balancing`),
#     que aplica SMOTE con `imbalanced-learn`.
#   - Abrir o regenerar los notebooks (`jupyter`, `nbformat`...).
#   - Subir runs y modelos a MLflow / DagsHub (`mlflow`).
#
# Las versiones respetan los mismos topes que `requirements.txt` para que
# instalar ambos a la vez no genere conflictos de resolución.

# --- Deep Learning (red neuronal con Keras) ---
# tensorflow-cpu solo publica ruedas para Linux y Windows; en macOS hay
# que usar el paquete `tensorflow`, que incluye builds para Intel y arm64.
tensorflow-cpu==2.16.2; sys_platform != "darwin"
tensorflow==2.16.2;     sys_platform == "darwin"
keras>=3.3,<3.6

# --- Balanceo de clases (bonus): SMOTE y similares ---
imbalanced-learn>=0.12,<0.14

# --- Visualización extra (gráficos en los notebooks y en `evaluator.py`) ---
seaborn>=0.13

# --- Notebooks (EDA, comparativa, modelos uno-a-uno) ---
jupyter>=1.0
ipykernel>=6.29
nbformat>=5.10
nbconvert>=7.16
notebook>=7.2

# --- MLflow (bonus): registro de experimentos y model registry ---
# Lo usamos contra DagsHub como servidor de tracking público; ver
# `docs/plan_despliegue_mlflow.md`.
mlflow>=2.16,<3.0
