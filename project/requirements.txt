# Dependencias del proyecto — Python 3.11 ó 3.12
# Instalación: pip install -r requirements.txt
#
# Las versiones se han elegido para que `pip install -r requirements.txt`
# funcione "out of the box" en las cuatro plataformas objetivo, sin pasos
# manuales:
#
#   - Linux  x86_64                (manylinux)
#   - Windows x86_64               (win_amd64)
#   - macOS Intel  (x86_64)
#   - macOS Apple Silicon (arm64)
#
# El techo lo fija TensorFlow 2.16.2, que es la última versión con rueda
# publicada para macOS x86_64. Esa cota a su vez obliga a numpy<2 y
# arrastra al resto de pines (pandas 2.x, scikit-learn 1.6.x, shap 0.49.x,
# etc.). En Linux/Windows usamos `tensorflow-cpu` (más ligero); en macOS
# el paquete `tensorflow` (única familia con ruedas para Intel + arm64).
#
# Python 3.13 todavía no tiene ruedas de TF 2.16, por eso el techo es 3.12.

# --- Núcleo de datos y ML clásico ---
numpy>=1.26,<2.0
pandas>=2.2,<3.0
scikit-learn>=1.5,<1.7
scipy>=1.13,<1.15
joblib>=1.4

# --- Gradient Boosting ---
xgboost>=2.1,<3.3

# --- Interpretabilidad (bonus): SHAP ---
shap>=0.46,<0.50

# --- Balanceo de clases (bonus): SMOTE y similares ---
imbalanced-learn>=0.12,<0.14

# --- Deep Learning (red neuronal con Keras) ---
# tensorflow-cpu solo publica ruedas para Linux y Windows; en macOS hay
# que usar el paquete `tensorflow`, que incluye builds para Intel y arm64.
tensorflow-cpu==2.16.2; sys_platform != "darwin"
tensorflow==2.16.2;     sys_platform == "darwin"
keras>=3.3,<3.6

# --- Visualización ---
matplotlib>=3.8,<3.11
seaborn>=0.13
plotly>=5.20

# --- Notebooks (EDA y comparativa) ---
jupyter>=1.0
ipykernel>=6.29
nbformat>=5.10
nbconvert>=7.16
notebook>=7.2

# --- API (bonus): FastAPI ---
fastapi>=0.110,<0.120
uvicorn[standard]>=0.30
requests>=2.32

# --- UI (bonus): Streamlit ---
streamlit>=1.36,<1.50
